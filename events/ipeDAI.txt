# Nam Tiên
country_event = {
	id = ipeDAI.1
	title = ipeDAI.1.t
	desc = ipeDAI.1.d
	picture = CITY_DEVELOPMENT_eventPicture

	trigger = {
		primary_culture = vietnamese
		any_owned_province = {
			OR = {
				area = vietnam_area
				area = southern_tonkin_area
				area = champa_area
				area = southern_champa_area
				area = tay_nguyen_area
				area = dong_nam_bo_area
				area = mekong_area
			}
			OR = {
				culture = cham
				culture = khmer
			}
		}
		is_at_war = no
		NOT = {
			has_country_flag = nam_tien_ends
		}
	}

	mean_time_to_happen = {
		months = 120
		modifier = {
			factor = 0.5
			has_country_modifier = enforcing_heritage
		}
	}

	option = { # But the march south goes on.
		name = ipeDAI.1.a
		ai_chance = {
			factor = 100
		}
		random_owned_province = {
			limit = {
				OR = {
					area = vietnam_area
					area = southern_tonkin_area
					area = champa_area
					area = southern_champa_area
					area = tay_nguyen_area
					area = dong_nam_bo_area
					area = mekong_area
				}
				OR = {
					culture = cham
					culture = khmer
				}
			}
			change_culture = vietnamese
			change_religion = root
		}
	}

	option = { # Put an end to the settlement of the south.
		name = ipeDAI.1.b
		ai_chance = {
			factor = 0
		}
		custom_tooltip = ipeDAI.1.b.tooltip
		set_country_flag = nam_tien_ends
	}
}

# Dynastic Crisis in Vietnam
country_event = {
	id = ipeDAI.2
	title = ipeDAI.2.t
	desc = ipeDAI.2.d
	picture = CIVIL_WAR_eventPicture

	fire_only_once = yes
	
	trigger = {
		is_emperor_of_china = yes
		OR = {
			exists = TOK
			exists = ANN
		}
		exists = DAI
		is_at_war = no
	}

	mean_time_to_happen = {
		months = 120
	}

	option = { # Support the claimants in Tonkin.
		name = ipeDAI.2.a
		trigger = {
			exists = TOK
		}
		TOK = {
			add_opinion = {
				who = ROOT
				modifier = supported_our_vietnamese_dynasty
			}
			country_event = {
				id = ipeDAI.3
				days = 1
			}
		}
		DAI = {
			add_opinion = {
				who = ROOT
				modifier = opposed_our_vietnamese_dynasty
			}
			country_event = {
				id = ipeDAI.4
				days = 1
			}

# The Death of Lê Bang Co
country_event = {
	id = ipeDAI.6
	title = ipeDAI.6.t
	desc = ipeDAI.6.d
	picture = DEATH_OF_HEIR_eventPicture

	fire_only_once = yes

	trigger = {
		tag = DAI
		has_ruler = "Thi Anh"
		dynasty = "Nguyên"
		has_ruler_flag = empress_dowager
		NOT = {
			has_heir = "Bang Co"
			has_country_flag = empress_dowager_dying
		}
		is_year = 1457
	}

	mean_time_to_happen = {
		days = 10
	}

	option = {
		name = ipeDAI.6.a
		ai_chance = {
			factor = 100
		}
		define_ruler = {
			name = "Tu Thành"
			dynasty = "Lê"
			adm = 5
			dip = 4
			mil = 5
			age = 20
			fixed = yes
		}
		hidden_effect = {
			add_ruler_personality = lawgiver_personality
		}
		if = {
			limit = {
				has_heir = yes
			}
			kill_heir = yes
		}
	}

	option = {
		name = ipeDAI.6.b
		ai_chance = {
			factor = 0
		}
		add_stability = -3
		capital_scope = {
			spawn_rebels = {
				type = pretender_rebels
				size = 1
				leader = "Tu Thành"
				leader_dynasty = "Lê"
			}
		}
		if = {
			limit = {
				religion_group = buddhist
			}
			set_karma = -100
		}
	}
}
