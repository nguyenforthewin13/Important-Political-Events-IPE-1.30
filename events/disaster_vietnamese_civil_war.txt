		if = {
			limit = {

			}

			name = "Dang Dung"
			dynasty = "Mac"

			age = 30
		}

			fire = 3
			shock = 3
			manuever = 3
			siege = 0
		}
		hidden_effect = {
			add_ruler_personality = malevolent_personality
		}
	}

		name = vietnamese_civil_war.2.b
		ai_chance = {
			factor = 50
		}
		set_country_flag = sided_with_the_trinh
		custom_tooltip = vietnamese_civil_war.2.b.tooltip
	}
}

# The Trinh-Nguyên Alliance
country_event = {
	id = vietnamese_civil_war.3
	title = vietnamese_civil_war.3.t
	desc = vietnamese_civil_war.3.d
	picture = BATTLE_eventPicture

	fire_only_once = yes

	major = yes

	trigger = {
		has_disaster = vietnamese_civil_war
		has_country_flag = sided_with_the_mac
		owns = 610
	}

	mean_time_to_happen = {
		months = 24
		modifier = {
			factor = 0.5
			stability = 1
		}
	}

	immediate = {
		clr_country_flag = sided_with_the_mac
		hidden_effect = {
			every_core_province = {
				limit = {
					OR = {
						area = vietnam_area
						area = southern_tonkin_area
						area = champa_area
						area = southern_champa_area
						area = tay_nguyen_area
						area = dong_nam_bo_area
						area = mekong_area
						area = south_laos_area
						area = cambodia_area
						area = eastern_cambodia_area
						area = cardomom_area
					}
				}
				add_core = ANN
			}
			release = ANN
			ANN = {
				change_government = monarchy
				add_government_reform = autocracy_reform
				set_government_rank = 2
				if = {
					limit = {
						owns = 610
					}
					set_capital = 610
				}
				capital_scope = {
					build_to_forcelimit = {
						infantry = 1.6
						cavalry = 0.4
					}
				}
				define_ruler = {
					name = "Kiêm"
					dynasty = "Trinh"
					adm = 2
					dip = 3
					mil = 3
					fixed = yes
				}
			}
			if = {
				limit = {
					is_subject_of_type = tributary_state
				}
				overlord = {
					create_subject = {
						subject_type = tributary_state
						subject = ANN
					}
				}
			}
		}
	}

	option = { # This upstart regime must be crushed!
		name = vietnamese_civil_war.3.a
		tooltip = {
			release = ANN
			ANN = {
				define_ruler = {
					name = "Kiêm"
					dynasty = "Trinh"
					adm = 2
					dip = 3
					mil = 3
					fixed = yes
				}
			}
		}
		ANN = {
			add_historical_rival = ROOT
		}
		add_historical_rival = ANN
		every_owned_province = {
			add_core = ANN
		}
	}

	option = { # Play as Annam.
		name = vietnamese_civil_war.3.b
		hidden_effect = {
			ANN = {
				add_historical_rival = ROOT
			}
			add_historical_rival = ANN
			every_owned_province = {
				add_core = ANN
			}
		}
		switch_tag = ANN
	}
}

# The Spreading of Famine
country_event = {
	id = vietnamese_civil_war.8
	title = vietnamese_civil_war.8.t
	desc = vietnamese_civil_war.8.d
	picture = SUFFERING_CHINESE_VILLAGE_eventPicture

	is_triggered_only = yes

	trigger = {
		any_owned_province = {
			base_tax = 2
			base_production = 2
			base_manpower = 2
		}
	}

	option = { # Heaven will one day smile upon us yet again.
		name = vietnamese_civil_war.8.a
		random_owned_province = {
			limit = {
				base_tax = 2
				base_production = 2
				base_manpower = 2
			}
			random_list = {
				33 = {
					add_base_tax = -1
				}
				33 = {
					add_base_production = -1
				}
				33 = {
					add_base_manpower = -1
				}
			}
			add_devastation = 10
		}
	}
}
