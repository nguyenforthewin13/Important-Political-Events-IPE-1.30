# The Treaty of Aláçovas
country_event = {
	id = ipeCAS.7
	title = ipeCAS.7.t
	desc = ipeCAS.7.d
	picture = DIPLOMACY_eventPicture

	fire_only_once = yes

	trigger = {
		tag = POR
		has_country_flag = canaries_rebel
		owns = 366
		controls = 366
		CAS = {
			exists = yes
		}
		is_at_war = no
	}

	mean_time_to_happen = {
		months = 120
	}

	option = {
		name = ipeCAS.7.a
		ai_chance = {
			factor = 95
		}
		366 = {
			cede_province = CAS
			remove_core = POR
		}
		if = {
			limit = {
				owns = 4565
			}
			4565 = {
				cede_province = CAS
				remove_core = POR
			}
		}
		CAS = {
			add_opinion = {
				who = POR
				modifier = treaty_of_alacovas
			}
			country_event = {
				id = ipeCAS.8
				days = 1
			}
		}
	}

	option = {
		name = ipeCAS.7.b
		ai_chance = {
			factor = 5
		}
		if = {
			limit = {
				owns = 4565
				4565 = {
					NOT = {
						is_core = CAS
					}
				}
			}
			4565 = {
				add_core = CAS
			}
		}
		add_casus_belli = {
			target = CAS
			type = cb_restore_personal_union
			months = 300
		}
		remove_historical_friend = CAS
		CAS = {
			remove_historical_friend = POR
			add_casus_belli = {
				target = POR
				type = cb_restore_personal_union
				months = 300
			}
			country_event = {
				id = ipeCAS.9
				days = 1
			}
		}
	}
}
