# The Traitorous Armagnacs
country_event = {
	id = ipeFRA.5
	title = ipeFRA.5.t
	desc = ipeFRA.5.d
	picture = ACCUSATION_eventPicture

	fire_only_once = yes

	major = yes

	trigger = {
		tag = FRA
		is_at_war = no
		AMG = {
			has_country_flag = fall_of_armagnac
			is_at_war = no
			is_subject_of_type = fief
			is_subject_of = FRA
		}
		NOT = {
			has_disaster = league_public_weal
		}
	}

	mean_time_to_happen = {
		months = 3
	}

	option = { # This is the end of the House of Armagnac!
		name = ipeFRA.5.a
		ai_chance = {
			factor = 90
			modifier = {
				factor = 100
				ruler_has_personality = cruel_personality
			}
			modifier = {
				factor = 100
				ruler_has_personality = malevolent_personality
			}
			modifier = {
				factor = 0
				ruler_has_personality = benevolent_personality
			}
			modifier = {
				factor = 0
				ruler_has_personality = kind_hearted_personality
			}
		}
		AMG = {
			kill_ruler = yes
			kill_heir = yes
		}
		inherit = AMG
		if = {
			limit = {
				NEM = {
					dynasty = "d'Armagnac"
					is_subject_of = FRA
					is_subject_of_type = french_fief
				}
			}
			NEM = {
				kill_ruler = yes
				kill_heir = yes
			}
			inherit = NEM
		}
		every_subject_country = {
			limit = {
				OR = {
					is_subject_of_type = french_fief
					is_subject_of_type = fief
				}
				NOT = {
					OR = {
						tag = AMG
						AND = {
							tag = NEM
							dynasty = "d'Armagnac"
						}
					}
				}
			}
			add_liberty_desire = -25
		}
	}

	option = { # I am a most merciful ruler.
		name = ipeFRA.5.b
		ai_chance = {
			factor = 10
			modifier = {
				factor = 0
				ruler_has_personality = cruel_personality
			}
			modifier = {
				factor = 0
				ruler_has_personality = malevolent_personality
			}
			modifier = {
				factor = 100
				ruler_has_personality = benevolent_personality
			}
			modifier = {
				factor = 100
				ruler_has_personality = kind_hearted_personality
			}
		}
		every_subject_country = {
			limit = {
				OR = {
					is_subject_of_type = french_fief
					is_subject_of_type = fief
				}
			}
			add_opinion = {
				who = FRA
				modifier = showed_mercy
			}
		}
	}
}
