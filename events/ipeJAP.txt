        trigger = {
            has_dlc = "Mandate of Heaven"
        }
        picture = SHOGUNATE_AUTHORITY_INCIDENT_eventPicture
    }

    is_triggered_only = yes

	immediate = {
        hidden_effect = {
			release = KMK
			if = {
				limit = {
					ASK = {
						owns = 3325
					}
				}
				3325 = {
					cede_province = KMK
				}
			}
            KMK = {
                capital_scope = {
                    build_to_forcelimit = {
                        infantry = 1
                        light_ship = 0.5
                        galley = 0.5
                    }
                }
                define_ruler = {
                    name = "Shigeuji"
                    dynasty = "Ashikaga"
                    adm = 3
                    dip = 5
                    mil = 2
                    fixed = yes
                }
				add_ruler_personality = malevolent_personality
				set_country_flag = growing_uesugi_power
				add_legitimacy = 50
            }
            ASK = {
                create_subject = {
                    subject_type = kubo_vassal
                    subject = KMK
                }
            }
            KMK = {
                change_government = monarchy
                add_government_reform = shogunate
                adopt_reform_progress = ROOT
				every_owned_province = {
					limit = {
						is_core = UES
					}
					remove_core = UES
				}
				every_owned_province = {
					limit = {
						is_core = ASK
					}
					remove_core = ASK
				}
            }
        }
	}

    option = { # Only time will tell what forgiveness might bring.
        name = ipeJAP.4.a
		KMK = {
			add_opinion = {
				who = UES
				modifier = kanto_kubo_appointed
			}
		}
		UES = {
			add_opinion = {
				who = KMK
				modifier = kanto_kubo_appointed
			}
		}
        create_alliance = KMK
    }

	option = { # Play as Kamakura.
		name = ipeJAP.4.b
		KMK = {
			add_opinion = {
				who = UES
				modifier = kanto_kubo_appointed
			}
		}
		UES = {
			add_opinion = {
				who = KMK
				modifier = kanto_kubo_appointed
			}
		}
        create_alliance = KMK
		switch_tag = KMK
	}
}
